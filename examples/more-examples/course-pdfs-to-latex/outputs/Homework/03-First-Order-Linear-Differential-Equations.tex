\textbf{In problems 1 and 2, find the general solution to the differential equations.}

\begin{enumerate}[leftmargin=*, label=\numbox{\arabic*}]

\item
    \begin{problem}[5cm]
        \((x^2 + 1)\dfrac{dy}{dx} + 2xy - 3x = 0\)
    \end{problem}

\item
    \begin{problem}[5cm]
        \(xy' + y = 4\sqrt{x}\,; \qquad x > 0.\)
    \end{problem}

\end{enumerate}

\textbf{In problems 3--7, solve the initial value problems.}

\begin{enumerate}[leftmargin=*, label=\numbox{\arabic*}, resume]

\item
    \begin{problem}[5cm]
        \(\dfrac{dy}{dx} = -2xy + 2x\,e^{(-x^2)}\,; \qquad y(0) = 2.\)
    \end{problem}

\item
    \begin{problem}[5cm]
        \(xy' = 2y + x^3 \sin(x)\,; \qquad y(1) = 0.\)
    \end{problem}

\item
    \begin{problem}[5cm]
        \((x^2 + 1)y' + 5xy = x\,; \qquad y(0) = 2.\)
    \end{problem}

\item
    \begin{problem}[5cm]
        \(xy' + 4y = \dfrac{e^{-x}}{x^3}\,; \qquad y(1) = 0,\ x > 0.\)
    \end{problem}

\item
    \begin{problem}[5cm]
        \((1 + x)y' + y = \cos(x)\,; \qquad y(0) = 1,\ 1 + x > 0.\)
    \end{problem}

\end{enumerate}
